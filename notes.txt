
// The five methods of adding elements

node.append(...args) inserts its arguments after the last child of the node
node.prepend(...args) inserts its arguments before the first child of the node
node.before(...args) inserts its arguments before the node
node.after(...args) inserts its arguments after the node
node.replaceWith(...args) replaces the node with the specified nodes or strings


// Unfolding an array of arguments

const list = document.querySelector(".todo-list");

// An array of tasks for today
const tasks = [
  "Do a project",
  "Walk the dog",
  "Complete a React tutorial"
];

// Convert the array of tasks for today to an array of elements
const taskElements = tasks.map(task => {
  const listItem = document.createElement("li");
  listItem.textContent = task;
  return listItem;
});

// Add all elements to the DOM by "unpacking"
// the array elements into append()
list.append(...taskElements);

/// Using <template>
If you add a <template> element to your HTML, 
it won’t be displayed on the website. 
But we can access these templates with JavaScript,
clone their content, create new nodes, and add them to the DOM.


<!-- This will not appear on the webpage -->
<template id="user-template">
  <div class="user">
    <img class="user__avatar" alt="avatar">
    <p class="user__name"></p>
  </div>
</template>



// Select the template and access its content.
const usersContainer = document.querySelector('.users-container');

const userTemplate = document
  .querySelector("#user-template")
  .content
  .querySelector(".user");

// Make a clone of the template’s content
const userElement = userTemplate.cloneNode(true);

// Add content to the clone
const userAvatar = userElement.querySelector(".user__avatar");

userAvatar.src = "https://placeholder.png";
const userName = userElement.querySelector(".user__name");
userName.textContent = "Laura Ipsum";

// Add the clone to the DOM
usersContainer.append(userElement);
